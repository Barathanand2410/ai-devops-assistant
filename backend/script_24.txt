```yaml
name: CI/CD Pipeline for LMS

on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Set up Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '16'

      - name: Install dependencies
        run: npm install

      - name: Run Lint
        run: npm run lint

      - name: Run Tests
        run: npm test

  deploy:
    runs-on: ubuntu-latest
    needs: build
    if: github.ref == 'refs/heads/main'
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Set up Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '16'

      - name: Build application
        run: npm run build

      - name: Deploy to Production
        env:
          NODE_ENV: production
          DEPLOY_KEY: ${{ secrets.DEPLOY_KEY }}
        run: |
          echo "${DEPLOY_KEY}" > deploy_key.pem
          chmod 600 deploy_key.pem
          npm run deploy -- --key deploy_key.pem
```
